  • 如果有很多组怎么办
levelValue = levels(factor(mpg$drv))
aveVec = rep(0, length(levelValue))
for(i in 1:length(levelValue)){
  aveVec[i] = mean(mpg[mpg$drv==levelValue[i],]$hwy)
}
